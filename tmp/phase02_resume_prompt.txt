2. Add a new CurriculumCallback(BaseCallback) class BELOW the existing SelfPlayCallback.
   It must have:
   - __init__(self, opponent_player, save_dir, swap_every, win_threshold=0.70, min_episodes=500, verbose=0)
   - Fields: _phase = "warmup", _win_window = deque(maxlen=min_episodes), _last_swap = 0, _swap_count = 0
   - _on_step(self) -> bool:
       * For each info dict in self.locals["infos"] where "episode" key exists:
           push (1 if info["episode"]["r"] > 0 else 0) into _win_window
       * if _phase == "warmup" and len(_win_window) >= min_episodes:
           win_rate = sum(_win_window) / len(_win_window)
           if win_rate >= win_threshold: call self._graduate()
       * if _phase == "selfplay" and num_timesteps - _last_swap >= swap_every:
           call self._save_and_swap(); _last_swap = num_timesteps
       * return True
   - _graduate(self):
       _swap_count += 1
       Save model to save_dir/swap_{_swap_count:04d}.zip, copy to save_dir/latest.zip
       Call opponent_player.load_policy(save_dir/"latest.zip")
       _phase = "selfplay"; _last_swap = num_timesteps
       Log graduation message
   - _save_and_swap(self):
       _swap_count += 1
       Save model to save_dir/swap_{_swap_count:04d}.zip, copy to save_dir/latest.zip
       Call opponent_player.load_policy(save_dir/"latest.zip")
       Log swap message
